(sno char(9) primary key,

(cno char(4) primary key,
 cname char(40),
 cpno char(4),
 ccredit smallint,

);

(sno char(9),
cno char(4),
grade smallint ,
primary key (sno,cno),
foreign key (sno) references student(sno),
foreign key (cno) references course(cno)
);
